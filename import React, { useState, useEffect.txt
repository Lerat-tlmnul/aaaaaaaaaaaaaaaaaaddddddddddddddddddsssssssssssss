    <script>
        // --- DATA STORE ---
        // On récupère les données du localStorage ou on initialise les valeurs par défaut
        const DB = {
            users: JSON.parse(localStorage.getItem('bm_users')) || [
                { user: 'Liam', pass: 'Liam', role: 'admin' },
                { user: 'Jules', pass: 'Jules', role: 'admin' }
            ],
            products: JSON.parse(localStorage.getItem('bm_products')) || [
                { id: 1, name: 'Kinder Bueno', price: 1.20, stock: 10 },
                { id: 2, name: 'Capri-Sun', price: 0.80, stock: 15 },
                { id: 3, name: 'Oreo Original', price: 1.00, stock: 8 }
            ],
            orders: JSON.parse(localStorage.getItem('bm_orders')) || []
        };

        let state = {
            currentUser: JSON.parse(localStorage.getItem('bm_session')) || null,
            cart: [],
            authMode: 'login'
        };

        // Fonction de sauvegarde globale
        const save = () => {
            localStorage.setItem('bm_users', JSON.stringify(DB.users));
            localStorage.setItem('bm_products', JSON.stringify(DB.products));
            localStorage.setItem('bm_orders', JSON.stringify(DB.orders));
            localStorage.setItem('bm_session', JSON.stringify(state.currentUser));
        };

        // --- AUTH ---
        function toggleAuth() {
            state.authMode = state.authMode === 'login' ? 'register' : 'login';
            document.getElementById('auth-title').innerText = state.authMode === 'login' ? 'Connexion' : 'Création';
            document.getElementById('btn-toggle-auth').innerText = state.authMode === 'login' ? 'Pas encore de compte ? Créer un profil' : 'Déjà un compte ? Se connecter';
        }

        function handleLogin() {
            const u = document.getElementById('login-user').value.trim();
            const p = document.getElementById('login-pass').value.trim();
            if(!u || !p) return notify('Champs vides', 'error');

            if(state.authMode === 'login') {
                const found = DB.users.find(x => x.user === u && x.pass === p);
                if(found) {
                    state.currentUser = found;
                    save();
                    notify(`Bienvenue ${u} !`);
                    initApp();
                } else {
                    notify('Identifiants incorrects', 'error');
                }
            } else {
                if(DB.users.find(x => x.user === u)) return notify('Pseudo déjà pris', 'error');
                DB.users.push({ user: u, pass: p, role: 'user' });
                save();
                notify('Compte créé ! Connectez-vous.');
                toggleAuth();
            }
        }

        function logout() {
            state.currentUser = null;
            state.cart = [];
            save();
            location.reload();
        }

        // --- VIEWS ---
        function showView(view) {
            ['view-auth', 'view-shop', 'view-orders', 'view-admin'].forEach(id => {
                document.getElementById(id).classList.add('hidden');
            });
            document.getElementById(`view-${view}`).classList.remove('hidden');
            
            if(view === 'shop') renderShop();
            if(view === 'orders') renderMyOrders();
            if(view === 'admin') renderAdmin();
        }

        function initApp() {
            if(!state.currentUser) return showView('auth');
            document.getElementById('nav').classList.remove('hidden');
            document.getElementById('nav-admin').classList.toggle('hidden', state.currentUser.role !== 'admin');
            showView('shop');
        }

        // --- SHOP & CART ---
        function renderShop() {
            const list = document.getElementById('product-list');
            list.innerHTML = DB.products.map(p => `
                <div class="bg-white dark:bg-slate-900 p-6 rounded-3xl border border-slate-100 dark:border-slate-800 shadow-sm admin-card">
                    <div class="h-12 w-12 bg-primary-50 dark:bg-primary-900/20 text-primary-600 rounded-2xl flex items-center justify-center mb-4">
                        <i class="fas fa-cookie-bite text-xl"></i>
                    </div>
                    <h3 class="font-black text-lg mb-1">${p.name}</h3>
                    <div class="flex items-center justify-between mb-6">
                        <span class="text-2xl font-black text-primary-600">${p.price.toFixed(2)}€</span>
                        <span class="text-[10px] font-black px-3 py-1 bg-slate-100 dark:bg-slate-800 rounded-full uppercase tracking-tighter">Stock: ${p.stock}</span>
                    </div>
                    <button onclick="addToCart(${p.id})" ${p.stock <= 0 ? 'disabled' : ''} class="w-full py-3 bg-slate-900 dark:bg-slate-800 text-white rounded-xl font-bold text-xs uppercase tracking-widest disabled:opacity-20 transition hover:bg-primary-600">
                        Ajouter
                    </button>
                </div>
            `).join('');
        }

        function addToCart(id) {
            const p = DB.products.find(x => x.id === id);
            const inCart = state.cart.find(x => x.id === id);
            if(inCart) {
                if(inCart.qty < p.stock) inCart.qty++;
                else return notify('Stock max atteint', 'warning');
            } else {
                state.cart.push({...p, qty: 1});
            }
            updateCartUI();
            notify(`${p.name} ajouté !`);
        }

        function updateCartUI() {
            const count = state.cart.reduce((a, b) => a + b.qty, 0);
            document.getElementById('cart-count').innerText = count;
            
            const total = state.cart.reduce((a, b) => a + (b.price * b.qty), 0) + 0.5;
            document.getElementById('cart-total').innerText = total.toFixed(2) + '€';

            document.getElementById('cart-items').innerHTML = state.cart.map(i => `
                <div class="flex items-center gap-4 p-4 bg-slate-50 dark:bg-slate-800/50 rounded-2xl">
                    <div class="flex-1">
                        <p class="font-black text-sm">${i.name}</p>
                        <p class="text-xs text-slate-400 font-bold">${i.price.toFixed(2)}€</p>
                    </div>
                    <div class="flex items-center gap-3">
                        <button onclick="changeQty(${i.id}, -1)" class="w-8 h-8 rounded-lg bg-white dark:bg-slate-800 shadow-sm flex items-center justify-center"><i class="fas fa-minus text-[10px]"></i></button>
                        <span class="font-black text-sm">${i.qty}</span>
                        <button onclick="changeQty(${i.id}, 1)" class="w-8 h-8 rounded-lg bg-white dark:bg-slate-800 shadow-sm flex items-center justify-center"><i class="fas fa-plus text-[10px]"></i></button>
                    </div>
                </div>
            `).join('');
        }

        function changeQty(id, delta) {
            const item = state.cart.find(x => x.id === id);
            const p = DB.products.find(x => x.id === id);
            item.qty += delta;
            if(item.qty > p.stock) item.qty = p.stock;
            if(item.qty <= 0) state.cart = state.cart.filter(x => x.id !== id);
            updateCartUI();
        }

        function showCart() { document.getElementById('cart-modal').classList.remove('hidden'); }
        function hideCart() { document.getElementById('cart-modal').classList.add('hidden'); }

        function checkout() {
            if(state.cart.length === 0) return;
            const spot = document.getElementById('delivery-spot').value;
            const total = state.cart.reduce((a, b) => a + (b.price * b.qty), 0) + 0.5;

            const order = {
                id: Date.now(),
                user: state.currentUser.user,
                items: [...state.cart],
                total: total,
                spot: spot,
                status: 'En attente',
                date: new Date().toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' })
            };

            // Diminuer stocks
            state.cart.forEach(item => {
                const p = DB.products.find(x => x.id === item.id);
                p.stock -= item.qty;
            });

            DB.orders.unshift(order);
            state.cart = [];
            save(); // Sauvegarde cruciale ici
            updateCartUI();
            hideCart();
            notify('Commande envoyée ! 🚀');
            showView('orders');
        }

        // --- MY ORDERS ---
        function renderMyOrders() {
            const list = document.getElementById('my-orders');
            const mine = DB.orders.filter(o => o.user === state.currentUser.user);
            if(mine.length === 0) {
                list.innerHTML = `<div class="p-10 text-center text-slate-400 font-bold border-2 border-dashed border-slate-200 dark:border-slate-800 rounded-3xl">Aucune commande en cours</div>`;
                return;
            }
            list.innerHTML = mine.map(o => `
                <div class="bg-white dark:bg-slate-900 p-6 rounded-3xl shadow-sm border border-slate-100 dark:border-slate-800">
                    <div class="flex justify-between items-center mb-4">
                        <span class="text-xs font-black text-primary-600 uppercase tracking-widest">${o.date}</span>
                        <span class="px-3 py-1 rounded-full text-[10px] font-black uppercase ${o.status === 'Livré' ? 'bg-green-100 text-green-600' : 'bg-primary-100 text-primary-600'}">${o.status}</span>
                    </div>
                    <div class="text-sm font-bold text-slate-500 mb-4">
                        ${o.items.map(i => `${i.qty}x ${i.name}`).join(', ')}
                    </div>
                    <div class="flex justify-between items-end border-t border-slate-50 dark:border-slate-800 pt-4">
                        <div class="text-[10px] font-black uppercase text-slate-400">📍 ${o.spot}</div>
                        <div class="text-xl font-black">${o.total.toFixed(2)}€</div>
                    </div>
                </div>
            `).join('');
        }

        // --- ADMIN ENGINE ---
        function renderAdmin() {
            // Inventaire
            const inv = document.getElementById('admin-inventory');
            inv.innerHTML = DB.products.map(p => `
                <div class="bg-white dark:bg-slate-900 p-5 rounded-2xl border border-slate-100 dark:border-slate-800 shadow-sm flex items-center justify-between">
                    <div class="flex-1">
                        <input type="text" onchange="adminUpdateProduct(${p.id}, 'name', this.value)" value="${p.name}" class="bg-transparent font-black outline-none focus:text-primary-500 block w-full">
                        <div class="flex items-center gap-4 mt-2">
                            <div class="flex flex-col">
                                <span class="text-[8px] font-black uppercase text-slate-400">Prix (€)</span>
                                <input type="number" step="0.01" onchange="adminUpdateProduct(${p.id}, 'price', this.value)" value="${p.price}" class="bg-transparent text-sm font-bold w-16 outline-none">
                            </div>
                            <div class="flex flex-col">
                                <span class="text-[8px] font-black uppercase text-slate-400">Stock</span>
                                <div class="flex items-center gap-2">
                                    <button onclick="adminAdjustStock(${p.id}, -1)" class="text-xs opacity-50"><i class="fas fa-minus"></i></button>
                                    <span class="text-sm font-bold">${p.stock}</span>
                                    <button onclick="adminAdjustStock(${p.id}, 1)" class="text-xs opacity-50"><i class="fas fa-plus"></i></button>
                                </div>
                            </div>
                        </div>
                    </div>
                    <button onclick="adminDeleteProduct(${p.id})" class="text-slate-300 hover:text-red-500 p-2 transition"><i class="fas fa-trash-alt"></i></button>
                </div>
            `).join('');

            // Commandes
            const flow = document.getElementById('admin-orders-flow');
            flow.innerHTML = DB.orders.map(o => `
                <div class="bg-white dark:bg-slate-900 p-5 rounded-2xl border border-slate-100 dark:border-slate-800 shadow-sm">
                    <div class="flex justify-between items-start mb-3">
                        <div>
                            <p class="text-xs font-black uppercase tracking-tighter">${o.user}</p>
                            <p class="text-[10px] text-slate-400 font-bold">${o.date} - ${o.spot}</p>
                        </div>
                        <span class="text-sm font-black">${o.total.toFixed(2)}€</span>
                    </div>
                    <div class="text-[11px] font-bold text-slate-500 mb-4">${o.items.map(i => `${i.qty}x ${i.name}`).join(', ')}</div>
                    <div class="flex gap-2">
                        ${o.status === 'En attente' ? `<button onclick="adminMarkOrder('${o.id}', 'Livré')" class="flex-1 py-2 bg-green-500 text-white text-[10px] font-black rounded-lg uppercase tracking-widest">Livrer</button>` : `<span class="flex-1 text-center py-2 text-green-500 text-[10px] font-black uppercase">Terminé</span>`}
                        <button onclick="adminDeleteOrder('${o.id}')" class="px-3 py-2 bg-slate-100 dark:bg-slate-800 rounded-lg text-slate-400 hover:text-red-500"><i class="fas fa-trash"></i></button>
                    </div>
                </div>
            `).join('');
        }

        function adminAddProduct() {
            const n = document.getElementById('new-p-name').value;
            const p = parseFloat(document.getElementById('new-p-price').value);
            const s = parseInt(document.getElementById('new-p-stock').value);
            if(!n || isNaN(p) || isNaN(s)) return notify('Informations incomplètes', 'error');

            DB.products.unshift({ id: Date.now(), name: n, price: p, stock: s });
            save(); // Sauvegarde immédiate
            renderAdmin();
            notify('Produit ajouté !');
            document.getElementById('new-p-name').value = '';
            document.getElementById('new-p-price').value = '';
            document.getElementById('new-p-stock').value = '';
        }

        function adminUpdateProduct(id, field, val) {
            const p = DB.products.find(x => x.id === id);
            if (!p) return;
            p[field] = field === 'price' ? parseFloat(val) : val;
            save(); // Sauvegarde immédiate du changement de prix ou nom
            notify('Modification enregistrée');
        }

        function adminAdjustStock(id, delta) {
            const p = DB.products.find(x => x.id === id);
            p.stock += delta;
            if(p.stock < 0) p.stock = 0;
            save(); // Sauvegarde immédiate du stock
            renderAdmin();
        }

        function adminDeleteProduct(id) {
            DB.products = DB.products.filter(x => x.id !== id);
            save(); // Sauvegarde immédiate après suppression
            renderAdmin();
            notify('Produit supprimé', 'warning');
        }

        function adminMarkOrder(id, status) {
            const o = DB.orders.find(x => x.id == id);
            o.status = status;
            save(); // Sauvegarde du statut de livraison
            renderAdmin();
            notify('Commande livrée !');
        }

        function adminDeleteOrder(id) {
            DB.orders = DB.orders.filter(x => x.id != id);
            save(); // Sauvegarde après suppression d'historique
            renderAdmin();
            notify('Historique supprimé');
        }

        // --- UTILS ---
        function notify(msg, type = 'success') {
            const n = document.createElement('div');
            n.className = `p-4 rounded-2xl shadow-2xl glass border-l-4 ${type === 'success' ? 'border-green-500 text-green-600' : (type === 'warning' ? 'border-orange-500 text-orange-600' : 'border-red-500 text-red-500')} flex items-center gap-3 animate-slide transform transition-all duration-300 pointer-events-auto`;
            n.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : 'exclamation-circle'}"></i> <span class="text-xs font-black uppercase tracking-widest">${msg}</span>`;
            document.getElementById('notifier').appendChild(n);
            setTimeout(() => { n.classList.add('opacity-0', 'translate-x-10'); setTimeout(() => n.remove(), 300); }, 2500);
        }

        window.onload = initApp;
    </script>
